134/// Function main() returns ExitCode.
135fn main() -> std::process::ExitCode {
136    match main_returns_anyhow_result() {
137        Err(err) => {
138            eprintln!("{}", err);
139            // eprintln!("Exit program with failure exit code 1");
140            std::process::ExitCode::FAILURE
141        }
142        Ok(()) => std::process::ExitCode::SUCCESS,
143    }
144}

146/// Function main() returns anyhow::Result.
147fn main_returns_anyhow_result() -> anyhow::Result<()> {
148    tracing_init()?;
149    msg_enc_dec_config_initialize().log(pos!())?;
150    // super simple argument parsing. There are crates that can parse more complex arguments.
151    match std::env::args().nth(1).as_deref() {
152        None | Some("--help") | Some("-h") => print_help().log(pos!())?,
153        // Register completion for msg_enc_dec  with the shell command 'complete -C'.
154        Some("register_completion") => register_completion().log(pos!())?,
155        // When registered completion calls msg_enc_dec, the first argument is the path of the program.
156        // Completion will react only on 'msg_enc_dec' as first word. Not ./msg_enc_dec or ~/msg_enc_dec,...
157        Some("msg_enc_dec") => msg_enc_dec_completion().log(pos!())?,
158        Some("create_ssh_key") => create_ssh_key().log(pos!())?,
159        Some("send_public_key") => send_public_key().log(pos!())?,
160        Some("receive_public_key") => receive_public_key().log(pos!())?,
161        Some("message_encrypt") => message_encrypt().log(pos!())?,
162        Some("message_decrypt") => message_decrypt().log(pos!())?,
163
164        Some("file_encrypt") => match std::env::args().nth(2).as_deref() {
165            // second argument
166            Some(file_name) => {
167                file_encrypt(file_name).log(pos!())?;
168            }
169            None => eprintln!("{RED}Error: Missing arguments `file_name`.{RESET}"),
170        },
171        Some("file_decrypt") => match std::env::args().nth(2).as_deref() {
172            // second argument
173            Some(file_name) => {
174                file_decrypt(file_name).log(pos!())?;
175            }
176            None => eprintln!("{RED}Error: Missing arguments `file_name`.{RESET}"),
177        },
178
179        _ => eprintln!("{RED}Error: Unrecognized arguments. Try `msg_enc_dec --help`{RESET}"),
180    }
181    Ok(())
182}

196/// Trait to log the error from Result before propagation with ?.
197pub trait ResultLogError<T, E>: Sized {
198    fn log(self, file_line_column: &str) -> Self;
199}

201/// Implements LogError for anyhow::Result.
202impl<T, E: std::fmt::Debug> ResultLogError<T, E> for core::result::Result<T, E> {
203    fn log(self, file_line_column: &str) -> Self {
204        self.inspect_err(|err| tracing::debug!("{} {:?}", file_line_column, err))
205    }
206}

208/// Initialize tracing to file logs/msg_enc_dec.log.  \
209///
210/// The folder logs/ is in .gitignore and will not be committed.  
211pub fn tracing_init() -> anyhow::Result<()> {
212    let offset = time::UtcOffset::current_local_offset()?;
213    let timer = tracing_subscriber::fmt::time::OffsetTime::new(
214        offset,
215        time::macros::format_description!("[hour]:[minute]:[second].[subsecond digits:6]"),
216    );
217
218    // A filter consists of one or more comma-separated directives
219    // target[span{field=value}]=level
220    // Levels order: 1. ERROR, 2. WARN, 3. INFO, 4. DEBUG, 5. TRACE
221    // ERROR level is always logged.
222    // Add filters to MSG_ENC_DEC_LOG environment variable for a single execution:
223    // ```bash
224    // MSG_ENC_DEC_LOG="debug,hyper_util=info,reqwest=info" ./{package_name}
225    // ```
226    let filter = tracing_subscriber::EnvFilter::from_env("MSG_ENC_DEC_LOG");
227
228    let builder = tracing_subscriber::fmt()
229        .with_file(true)
230        .with_timer(timer)
231        .with_line_number(true)
232        .with_ansi(false)
233        .with_env_filter(filter);
234    if std::env::var("MSG_ENC_DEC_LOG").is_ok() {
235        // if MSG_ENC_DEC_LOG exists than enable tracing to file
236        let file_appender = tracing_appender::rolling::RollingFileAppender::builder()
237            .rotation(tracing_appender::rolling::Rotation::DAILY)
238            .filename_prefix("msg_enc_dec")
239            .filename_suffix("log")
240            .build("logs")
241            .expect("initializing rolling file appender failed");
242        builder.with_writer(file_appender).init();
243    } else {
244        builder.init();
245    };
246
247    Ok(())
248}

363/// Print one or more sub_commands.
364pub fn completion_return_one_or_more_sub_commands(sub_commands: Vec<&str>, word_being_completed: &str) {
365    let mut sub_found = false;
366    for sub_command in sub_commands.iter() {
367        if sub_command.starts_with(word_being_completed) {
368            println!("{sub_command}");
369            sub_found = true;
370        }
371    }
372    if !sub_found {
373        // print all sub-commands
374        for sub_command in sub_commands.iter() {
375            println!("{sub_command}");
376        }
377    }
378}

380/// Create ssh key and config json.
381fn create_ssh_key() -> anyhow::Result<()> {
382    // instead of using ssh-keygen, I will create the private and public key in Rust code using ed25519-dalek
383
384    println!("  {YELLOW}Generate the ssh private/public key pair. {RESET}");
385    println!("  {YELLOW}Give it a good passphrase and remember it. {RESET}");
386    println!("  {YELLOW}Nobody can help you if you forget the passphrase. {RESET}");
387    println!("  {YELLOW}You would have to delete the old key and create a new one. {RESET}");
388    println!("  {YELLOW}This ssh key will be used to save locally the secret session token for the communication. {RESET}");
389    let secret_passphrase: String = dialoguer::Password::new()
390        .with_prompt(format!("{BLUE}Write secret passphrase{RESET}"))
391        .interact()
392        .log(pos!())?;
393
394    let private_key = ssh_key::PrivateKey::random(&mut aes_gcm::aead::OsRng, ssh_key::Algorithm::Ed25519).log(pos!())?;
395    let encrypted_key = private_key.encrypt(&mut aes_gcm::aead::OsRng, secret_passphrase).log(pos!())?;
396    let path = CrossPathBuf::new("msg_enc_dec_ssh_1").log(pos!())?;
397    encrypted_key
398        .write_openssh_file(&path.to_path_buf_current_os(), ssh_key::LineEnding::LF)
399        .log(pos!())?;
400    let path = CrossPathBuf::new("msg_enc_dec_ssh_1.pub").log(pos!())?;
401    private_key
402        .public_key()
403        .write_openssh_file(&path.to_path_buf_current_os())
404        .log(pos!())?;
405
406    println!();
407    println!("  {YELLOW}After create_ssh_key run 'msg_enc_dec send_public_key'. {RESET}");
408    Ok(())
409}

411/// Print the static public key to be sent.
412fn send_public_key() -> anyhow::Result<()> {
413    // https://docs.rs/crate/x25519-dalek/
414    // create static secret, because ephemeral secrets cannot be extracted and eaved to file.
415    let static_secret: x25519_dalek::StaticSecret = x25519_dalek::StaticSecret::random();
416
417    // Save the static secret encrypted into local folder.
418    let private_key_file_name = global_private_key_file_name().log(pos!())?;
419    let encrypted_secret_file_path = CrossPathBuf::new("static_secret.enc").log(pos!())?;
420    encrypt_and_save_file(&private_key_file_name, static_secret.to_bytes(), &encrypted_secret_file_path).log(pos!())?;
421
422    // Send the public key to the other party.
423    let public_key = x25519_dalek::PublicKey::from(&static_secret);
424    let public_key_string = ende::encode64_from_bytes_to_string(public_key.to_bytes().to_vec());
425
426    println!("  {YELLOW}Send this public key to the other party. This is not a secret. {RESET}");
427    println!("  {YELLOW}They must use 'msg_enc_dec receive_public_key'. {RESET}");
428    println!("  {YELLOW}and then send you the encrypted session token. {RESET}");
429    println!("  {YELLOW}It is encrypted, only the owner of the private key can decrypt it. {RESET}");
430    println!(r#"{GREEN}{public_key_string} {RESET}"#);
431
432    Ok(())
433}

435/// Save the secret bytes symmetrically encrypted into a file.
436///
437/// Use the private key to sign the random seed. The random seed is saved as plain inside the file.
438/// The file is bas64 only to masquerade it a little bit.
439fn encrypt_and_save_file(
440    private_key_file_name: &str,
441    secret_bytes: [u8; 32],
442    encrypted_secret_file_path: &CrossPathBuf,
443) -> Result<(), anyhow::Error> {
444    let (plain_seed_bytes_32bytes, plain_seed_string) = ende::random_seed_32bytes_and_string().log(pos!())?;
445    let private_key_path = CrossPathBuf::new(private_key_file_name).log(pos!())?;
446    let secret_passcode_32bytes: SecretBox<[u8; 32]> =
447        ende::sign_seed_with_ssh_agent_or_private_key_file(&private_key_path, plain_seed_bytes_32bytes).log(pos!())?;
448    let secret_string = secrecy::SecretString::from(ende::encode64_from_32bytes_to_string(secret_bytes).log(pos!())?);
449    debug!("secret_string: {}", secret_string.expose_secret());
450    let plain_encrypted_text = ende::encrypt_symmetric(secret_passcode_32bytes, secret_string).log(pos!())?;
451    let json_struct = EncryptedTextWithMetadata {
452        plain_seed_string,
453        plain_encrypted_text,
454    };
455    let json_string = serde_json::to_string_pretty(&json_struct).log(pos!())?;
456    encrypted_secret_file_path.write_str_to_file(&json_string).log(pos!())?;
457    Ok(())
458}

460/// Receive public key, calculate shared secret, encrypt and store for later use.
461fn receive_public_key() -> anyhow::Result<()> {
462    let other_public_key: String = dialoguer::Input::new()
463        .with_prompt(format!("{BLUE}Copy the public key received from the other party{RESET}"))
464        .interact_text()
465        .log(pos!())?;
466    let other_public_key = ende::decode64_from_string_to_32bytes(&other_public_key).log(pos!())?;
467    let other_public_key = x25519_dalek::PublicKey::from(other_public_key);
468
469    // load and decrypt the static secret
470    let private_key_file_name = global_private_key_file_name().log(pos!())?;
471    let enc_file_path = CrossPathBuf::new("static_secret.enc").log(pos!())?;
472    let static_secret_bytes = load_and_decrypt(&private_key_file_name, &enc_file_path).log(pos!())?;
473    let static_secret = x25519_dalek::StaticSecret::from(static_secret_bytes);
474
475    // calculate shared secret
476    let shared_secret = static_secret.diffie_hellman(&other_public_key);
477
478    // save encrypted shared secret
479    let private_key_file_name = global_private_key_file_name().log(pos!())?;
480    let encrypted_secret_file_path = CrossPathBuf::new("shared_secret.enc").log(pos!())?;
481    encrypt_and_save_file(&private_key_file_name, shared_secret.to_bytes(), &encrypted_secret_file_path).log(pos!())?;
482
483    // for debugging I can write the encrypted session token that is created
484    // let session_token_enc_path = CrossPathBuf::new("enc_session_token_1.txt").log(pos!())?;
485    // session_token_enc_path.write_str_to_file(&plain_session_token).log(pos!())?;
486
487    println!("  {YELLOW}The shared secret session token is saved.{RESET}");
488    println!("  {YELLOW}Now you can encrypt and decrypt messages and files.{RESET}");
489    println!(r#"{GREEN}msg_enc_dec message_encrypt {RESET}"#);
490    println!(r#"{GREEN}msg_enc_dec message_decrypt {RESET}"#);
491    println!(r#"{GREEN}msg_enc_dec file_encrypt file_name {RESET}"#);
492    println!(r#"{GREEN}msg_enc_dec file_decrypt file_name {RESET}"#);
493
494    Ok(())
495}

497// Load and decrypt secret.
498fn load_and_decrypt(private_key_file_name: &str, encrypted_secret_file_path: &CrossPathBuf) -> Result<[u8; 32], anyhow::Error> {
499    let encrypted_secret_file_string = encrypted_secret_file_path.read_to_string().log(pos!())?;
500    let json_struct: EncryptedTextWithMetadata = serde_json::from_str(&encrypted_secret_file_string).log(pos!())?;
501    let plain_seed_bytes_32bytes = ende::decode64_from_string_to_32bytes(&json_struct.plain_seed_string).log(pos!())?;
502    let private_key_path = CrossPathBuf::new(private_key_file_name).log(pos!())?;
503    let secret_passcode_32bytes: SecretBox<[u8; 32]> =
504        ende::sign_seed_with_ssh_agent_or_private_key_file(&private_key_path, plain_seed_bytes_32bytes).log(pos!())?;
505    let secret_string = ende::decrypt_symmetric(secret_passcode_32bytes, json_struct.plain_encrypted_text).log(pos!())?;
506    let secret_bytes = ende::decode64_from_string_to_32bytes(secret_string.expose_secret()).log(pos!())?;
507    Ok(secret_bytes)
508}
